on b.stvdegc_code = a.sordegr_degc_code

join so1degr c
on c.sordegr_pidm = a.sordegr_pidm
and c.sordegr_degc_code = a.sordegr_degc_code
and c.sordegr_sbgi_code = a.sordegr_sbgi_code
and c.sordegr_degr_seq_no = a.sordegr_degr_seq_no
left
join saturn.sorpcol w
  on w.sorpcol_pidm      = a.sordegr_pidm
and    w.sorpcol_sbgi_code = a.sordegr_sbgi_code
left
join so2cees d
on a.sordegr_pidm = d.sordegr_pidm
and a.sordegr_sbgi_code = d.sordegr_sbgi_code
and a.sordegr_degc_code = d.sordegr_degc_code
and a.sordegr_degr_seq_no = d.sordegr_degr_seq_no
and d.so2cees_stts_cert = 'C'
where b.stvdegc_dlev_code >= '05'
union all
